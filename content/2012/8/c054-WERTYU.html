解題雜記：
　　這題不論是用if-else、switch-case或是把對應的字元先存在陣列中，這三種做法都可能讓人有種打字打到崩潰的感覺。為了節省打字的功天，這邊的做法是把鍵盤上所有的字元先打成字串存起來。然後每當有一個字元輸入，就找找看該字元在字串中的哪個位置，然後輸出字串中該位置-1的字元就結束了。

import java.util.Scanner;

public class d054 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  String keyboard = "`1234567890-=QWERTYUIOP[]\\ASDFGHJKL;'ZXCVBNM,./";
  
  while(input.hasNext()){
   char[] str = input.nextLine().toCharArray();
   
   for(int i = 0; i < str.length; ++i){
    for(int j = 0; j < keyboard.length(); ++j){
     if(str[i] == keyboard.charAt(j)){
      str[i] = keyboard.charAt(j - 1);
     }
    }
   }
   
   System.out.println(new String(str));
  }
 }

}
