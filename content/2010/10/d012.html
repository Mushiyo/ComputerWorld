#include <iostream>
#include <string>
using namespace std;

bool isPalindrome(string s);

int main()
{
    string str;
    while(cin >> str){
        if(isPalindrome(str)){
            cout << "Yes\n";
        }
        else{
            cout << "No\n";
        }
    }
    return 0;
}

bool isPalindrome(string s)
{
    int length = s.length();
    
    
    if (length == 1){
        return true;
    }
    else if(length == 2){
        if(s.at(0) == s.at(1)){
            return true;
        }
        else{
            return false;
        }
    }
    else if(s.at(0) == s.at(length - 1)){
        s = s.substr (1,length - 2);
        return isPalindrome(s);
    }
    else{
        return false;
    }
        
}
