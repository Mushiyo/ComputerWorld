#include <stdio.h>


int main(){
    int i, j, k;
    int count;
    int maxLength = 1, currentLength = 1;

    while(scanf("%d%d", &i, &j) == 2){

        printf("%d %d", i, j);
        k = i;
        if(i > j){
            k = j;
            j = i;
            i = k;
        }

        for(count = i; count <= j; count++, i++){
            while(k != 1){
                if(k % 2){
                    k = 3 * k + 1;
                }
                else{
                    k /= 2;
                }
                currentLength ++;
            }
            k = i + 1;

            if (maxLength < currentLength){
                maxLength = currentLength;
            }
            currentLength = 1;
        }
        printf(" %d\n", maxLength);
        maxLength = 1;
    }

    return 0;

}
