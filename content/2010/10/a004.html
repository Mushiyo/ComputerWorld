#include <stdio.h>

int main(){
    int year;
    while(scanf("%d",&year)==1){
        if( !(year % 4) && year % 100 || !(year % 400)){
            printf("閏年\n");
        }
        else{
            printf("平年\n");
        }
    }

    return 0;
}
