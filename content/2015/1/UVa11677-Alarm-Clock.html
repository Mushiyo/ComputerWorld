Solution Description:
This is an easy problem. The answer is H2 * 60 + M2 - (H1 * 60 + M1). If the answer is negative, add 24 * 60. The following is different from this description but the idea behind it is the same.

Java：
/* Filename: UVa11677.java
 * Author: Mushiyo
 */

import java.util.Scanner;

public class UVa11677 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);

  while (input.hasNext()) {
   int H1 = input.nextInt();
   int M1 = input.nextInt();
   int H2 = input.nextInt();
   int M2 = input.nextInt();
   
   if(H1 == 0 && M1 == 0 && H2 == 0 && M2 == 0){
    break;
   }
   
   if(H2 < H1){
    H2 += 24;
   }
   
   if(H2 == H1 && M2 < M1){
    H2 += 24;
   }
   
   int sleepMinute = 60 * H2 + M2 - (60 * H1 + M1);
   
   System.out.println(sleepMinute);
  }
 }

}
