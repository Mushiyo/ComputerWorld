Solution Description:
From the problem statement, we know that
  1. a1 is the weight of mathematician1's basket + berries, a3 is the weight of mathematician1's basket
  2. b1 is the weight of mathematician2's basket + berries, b2 is the weight of mathematician2's basket
Hence,  a1 - a3 is the weight of mathematician1's berries and b1 - b2 is the weight of mathematician2's berries

Java：
/* Filename: Timus2001.java
 * Author: Mushiyo
 */

import java.util.Scanner;

public class Timus2001 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  int[] a = new int[4];
  int[] b = new int[4];

  while (input.hasNext()) {
   for(int i = 1; i < a.length; ++i){
    a[i] = input.nextInt();
    b[i] = input.nextInt();
   }
   
   System.out.printf("%d %d\n", a[1] - a[3], b[1] - b[2]);
  }
 }

}
