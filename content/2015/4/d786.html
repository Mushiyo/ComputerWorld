解題雜記：
　　基本題。直接照題目要求做就好。記得在算平均值的時候，除數或被除數有一者要是double，不然若二者都是int的話，不管到底除不除得盡，算出來的結果都不會有小數點。

Java：
/* Filename: d786.java
 * Author: Mushiyo
 */
package BASIC;

import java.util.Scanner;

public class d786 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);

  while (input.hasNext()) {
   int totalCase = input.nextInt();
   
   while(totalCase > 0){
    int n = input.nextInt();
    
    int sum = 0;
    for(int i = 0; i < n; ++i){
     sum += input.nextInt();
    }
    
    System.out.printf("%.2f\n", sum / (double)n);
    
    --totalCase;
   }
  }
 }

}
