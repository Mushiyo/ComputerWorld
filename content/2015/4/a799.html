Java：
/* Filename: a799.java
 * Author: Mushiyo

package BASIC;

import java.util.Scanner;

public class a799 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);

  while (input.hasNext()) {
   System.out.println(Math.abs(input.nextInt()));
  }
 }

}
