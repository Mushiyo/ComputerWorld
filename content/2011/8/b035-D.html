#include <stdio.h>

int main(){
    int n = 0;
    int conqueredHour = 0;
    int sideLength = 0;
    int i = 1;
 
    scanf("%d", &n);
 
    while(n != 0){
        for(i = 1; i <= n; ++i) {
            scanf("%d", &sideLength);
            conqueredHour += (sideLength * sideLength);
        }
        printf("%d\n", conqueredHour);
  
        conqueredHour = 0;
        scanf("%d", &n);
    }
    
    return 0;
}
