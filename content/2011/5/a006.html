#include <stdio.h>
#include <math.h>

int main(){
    double a, b, c;
    double d;
    double x1, x2;
    
    while(scanf("%lf %lf %lf", &a, &b, &c) == 3){
        d = b * b - 4 * a * c;

        if(d == 0){
            x1 = (- b) / (2 * a);
            if(x1 == -0){
                x1 = 0;
            }
            printf("Two same roots x=%.0lf\n", x1);
        }
        else if(d < 0){
            printf("No real root\n");
        }
        else{
            x1 = (- b + sqrt(d)) / (2 * a);
            x2 = (- b - sqrt(d)) / (2 * a);
            printf("Two different roots x1=%.0lf , x2=%.0lf\n", x1, x2); 
        }
    }

    return 0;
}
