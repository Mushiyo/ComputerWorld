解題雜記：
　　觀察題目中對quirksome number的定義，我們可以發現當一個數是quirksome number，則它也一定完全平方數。因此若要找出所N位數的quirksome number，只要檢查1、4、9……(10^(N/2) - 1) * (10^(N/2) - 1)這幾個數是不是quirksome number就好，不需要枚舉出所有1 ~ ((10 ^ N) - 1)來檢查（事實上，根據某人的親身試驗，枚舉出所有的數會TLE……）。

package UVa;

import java.util.Scanner;

public class d182 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  
  while(input.hasNext()){
   int N = input.nextInt();
   //int upperBound = (int)Math.pow(10, N);
   int halfBound = (int)Math.pow(10, N / 2);
   
   for(int i = 0; i < halfBound; ++i){
    int squareI = i * i;
    int leftHalf = squareI / halfBound;
    int rightHalf = squareI % halfBound;
    
    if(Math.pow(leftHalf + rightHalf, 2) == squareI){
     System.out.println(String.format("%" + N + "d", squareI).replace(' ', '0'));
    }
   }
   
  }
 }

}
