解題雜記：
　　經由仔細觀察，可發現磚牆存在遞回關係。令p(n)表示長度為n的磚牆的花樣種類，則p(1) = 1，p(2) = 2， p(n) = p(n - 1) + p(n - 2)，n ≧ 3。先建表算出所有可能的p(n)，再根據測資的要求，一一輸出對應p(n)就行了。

package UVa;

import java.util.Scanner;

public class d038 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  long[] lenPattern = new long [51];
  lenPattern[1] = 1;
  lenPattern[2] = 2;
  for(int i = 3; i < lenPattern.length; ++i){
   lenPattern[i] = lenPattern[i - 1] + lenPattern[i - 2];
  }
  
  
  while(input.hasNext()){
   int length = input.nextInt();
   
   if(length == 0){
    break;
   }
   
   System.out.println(lenPattern[length]);
  }
 }

}
