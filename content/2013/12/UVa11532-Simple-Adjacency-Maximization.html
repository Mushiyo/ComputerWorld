解題雜記：
　　性質一告訴我們：要盡可能的把1往右擺，把0往左擺（這樣造出來的數字才會是最小的）。性質二告訴我們：要盡可能造出很多個"101"並把它們都串接在一起，即
101 101 101 ...這樣的形式。造完之後，如果還有0剩下來的話，此時若有剩餘的1（若有，一定只會剩一個），就把它放到放才造出的(101)*的左邊，即1(101)*這樣的形式，即為最佳解（1.剩的0都放到最左邊，為leading zero，故可忽略。 2.這樣才有符合性質二「與0相接的1要盡可能的多」）；若沒有0剩下來的話，就把所有的1都放到(101)*右邊，即(101)*1*這樣的形式，即為最佳解。

import java.util.Scanner;

public class UVa11532 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);

  while (input.hasNext()) {
   int C = input.nextInt();
   
   while (C > 0) {
    int P = input.nextInt();
    int Q = input.nextInt();
    StringBuilder bitNum = new StringBuilder();

    while (P >= 2 && Q > 0) {
     bitNum.append("101");
     P -= 2;
     --Q;
    }

    if (P > 0 && Q > 0) {
     bitNum.insert(0, 1);
     --P;
     --Q;
    }

    while (P > 0) {
     bitNum.append(1);
     --P;
    }

    bitNum.insert(0, 0);

    System.out.println(Long.valueOf(bitNum.toString(), 2));

    --C;
   }
  }

 }
}
