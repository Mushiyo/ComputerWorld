解題雜記：
　　由高中物理，我們可以知道fastest path和shortest path的情形分別如下圖中的(1)、(2)所示，因此可知fastest path所需的時間為，shortest path所需的時間為。接下來需要判斷是否確實存在兩條相同路徑。由前面的結果，可以看出：
1. 當v = 0時，fastest path = shortest path，實際上只有一條路徑
2. 當時，shortest path不存在
3. 當u = 0時，fastest path和shortest path皆不存在
　　因為題目告訴我們，所以可知u = 0這個case其實是包含在這個case中的，故只要就v = 0跟這種情況做判斷即可。

/* Filename: UVa10773.java

import java.util.Scanner;

public class UVa10773 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  int testCase = input.nextInt();

  for (int t = 1; t <= testCase; ++t) {
   double d = input.nextDouble();
   double v = input.nextDouble();
   double u = input.nextDouble();

   if (u <= v || v == 0) {
    System.out.println("Case " + t + ": can't determine");
   } else {
    System.out.printf("Case %d: %.3f\n", t, countDiff(d, v, u));
   }

  }

 }

 public static double countDiff(double d, double v, double u) {
  return d / Math.sqrt(u * u - v * v) - d / u;
 }
}

#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

double countDiff(double d, double v, double u) {
 return d / sqrt(u * u - v * v) - d / u;
}

int main() {
 int testCase = 0;
 cin >> testCase;

 for (int t = 1; t <= testCase; ++t) {
  double d, v, u;
  cin >> d >> v >> u;

  if (u <= v || v == 0) {
   printf("Case %d: can't determine\n", t);
  } else {
   printf("Case %d: %.3f\n", t, countDiff(d, v, u));
  }
 }

 return 0;
