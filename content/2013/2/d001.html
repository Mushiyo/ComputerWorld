解題雜記：
　　將字串依倒序輸出即可。

C++： 
#include <iostream>
#include <string>

using namespace std;

int main(){
 string s;
 
 while(getline(cin, s)){
  for(string::reverse_iterator rit = s.rbegin(); rit != s.rend(); ++rit){
   cout << *rit;
  }
  cout << endl;
 }

 return 0;
}
