解題雜記：
　　經觀察可發現f(n) = n + (n - 1) + (n - 2) + … + 2 + 1，故用梯形公式就可算出。
而g(n) = f(n) + f(n - 1) + f(n - 2) + … + f(2) + f(1)
　　　=  n * 1 + (n - 1) * 2 + (n - 2) * 3 + …… + (n - (n - 2)) * (n - 1) + (n - (n - 1)) * n
　　　=  n * 1 + (n - 1) * 2 + (n - 2) * 3 + …… + 2 * (n - 1) + 1 * n
因為我推不出公式，所以直接跑迴圈算。理論上公式應當推得出來，有與趣的人不妨試試！

package BASIC;

import java.util.Scanner;

public class a216 {

 public static void main(String[] args) {
  Scanner input = new Scanner(System.in);
  
  while(input.hasNext()){
   int n = input.nextInt();
   
   System.out.println(f(n) + " " + g(n));
  }
 }
 
 public static int f(int n){
  return ((1 + n) * n) / 2;
 }
 
 public static long g(int n){
  long sum = 0;
  
  for(int i = 1 ; i <= n; ++i){
   sum += ((n + 1) - i) * i;
  }
  
  return sum;
 }

}
