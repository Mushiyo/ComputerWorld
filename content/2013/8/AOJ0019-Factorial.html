解題雜記：
　　由於題目中有限定n ≦ 20，所以不用擔心溢位的問題。勇敢的從1乘到n就好:D

C++： 
/*Filename: aoj0019.cpp
 *Author: Mushiyo
 */

#include <iostream>

using namespace std;

long long factorial(int n){
 long long result = 1;

 for(int i = 1; i <= n; ++i){
  result *= i;
 }

 return result;
}

int main(){
 int n = 0;

    while(cin >> n){
        cout << factorial(n) << endl;
    }

 return 0;
}
