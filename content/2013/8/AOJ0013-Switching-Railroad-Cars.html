解題雜記：
　　用stack來存放進鐵軌的列車。當輸入是非0的數字時，就把數字壓入stack；當輸入是0時，就把stack最頂端的數字pop掉。如此模擬即可。

C++： 
/*Filename: aoj0013.cpp
 *Author: Mushiyo
 */

#include <iostream>
#include <stack>

using namespace std;

int main(){
 int num = 0;
 stack<int> car;

    while(cin >> num){
        if(num == 0){
         cout << car.top() << endl;
         car.pop();
        }
        else{
         car.push(num);
        }
    }

 return 0;
}
