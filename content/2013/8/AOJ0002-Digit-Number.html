解題雜記：
　　對a+b的和取log，把小數的部份無條件捨去後再加1，即為所求。要注意的是，由於我們不能對0取log，所以當和是0的時候要特別處理。

C++： 
/*Filename: aoj0002.cpp
 *Author: Mushiyo
 */

#include <iostream>
#include <cmath>

using namespace std;

int countDigit(int num){
 if(num == 0){
  return 1;
 }

 int digit = (int)log10(num) + 1;

 return digit;
}

int main(){
 int a = 0, b = 0;

    while(cin >> a >> b){
        cout << countDigit(a + b) << endl;
    }

 return 0;
}
