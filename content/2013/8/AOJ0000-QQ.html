解題雜記：
　　用雙層for迴圈，一行一行的輸出99乘法即可。

C++： 
/*Filename: aoj0000.cpp
 *Author: Mushiyo
 */

#include <iostream>

using namespace std;

int main(){

 for(int i = 1; i <=9; ++i){
  for(int j = 1; j <= 9; ++j){
   cout << i << "x" << j << "=" << i * j << endl;
  }
 }

 return 0;
}
