解題雜記：
　　這題要利用getline來把字串切成一個個的token，再判斷這些token是數字還是字串來決定要不要反向輸出。由於getline用來切字串的分隔符號只能設一個，所以前面用了istringstream先拿掉前後的括號，再做為輸入送給getline來切字串。

C++： 
/*Filename: d074.cpp
 *Author: Mushiyo
 */

#include <iostream>
#include <sstream>
#include <string>
#include <cctype>

using namespace std;

int main() {
 string s;

 while (getline(cin, s)) {
  istringstream iss(s.substr(1, s.size() - 2));
  string token;

  bool isFirstToken = true;
  cout << "{";
  while (getline(iss, token, ',')) {
   if (isFirstToken == false) {
    cout << ",";
   } else {
    isFirstToken = false;
   }

   if (isdigit(token[0])) {
    cout << token;
   } else {
    for (int i = token.size() - 1; i >= 0; --i) {
     cout << token[i];
    }
   }
  }
  cout << "}" << endl;

 }

 return 0;
}
