解題雜記：
　　用stack來裝由A方向開進站的車廂，然後用一個數字（底下的程式碼用的是carriageInB，不過這邊為了方便，使用i來稱呼這個數字）記錄目前有多少個車廂進到B方向的鐵軌。檢查stack的top是不是第i個要進到B的車廂，如是的話就pop掉（表示進到B方向鐵軌了）；如果不是的話，從A方向再開一節車廂進來（push進stack）。重覆這樣的動作直到A方向沒車廂了。此時檢查stack是否為空，如果是的話，表示可以排出要求的排列；不是的話就不行。

C++： 
/*Filename: d044.cpp
 *Author: Mushiyo
 */

#include <iostream>
#include <stack>

using namespace std;

int main() {
 int N = 0;

 while (cin >> N) {
  if (N == 0) {
   break;
  }

  while (true) {
   int firstCarriage = 0;
   cin >> firstCarriage;

   if (firstCarriage == 0) {
    break;
   }

   int targetOrder[N];
   targetOrder[0] = firstCarriage;
   for (int i = 1; i < N; ++i) {
    cin >> targetOrder[i];
   }

   int carriageInB = 0;
   stack<int> carriageLeaveA;
   for (int i = 1; i <= N; ++i) {
    carriageLeaveA.push(i);

    while (!carriageLeaveA.empty()
      && targetOrder[carriageInB] == carriageLeaveA.top()) {
     carriageLeaveA.pop();
     ++carriageInB;
    }
   }

   if (carriageInB == N) {
    cout << "Yes" << endl;
   } else {
    cout << "No" << endl;
   }
  }
  cout << endl;
 }
}
