解題雜記：
　　  倒序將輸入的字串輸出即可。

C++： 
/*Filename: d211.cpp
 *Author: Mushiyo
 */

#include <iostream>
#include <string>

using namespace std;

int main(){
 string s;

    while(getline(cin, s)){
        for(int i = s.size() - 1; i >= 0; --i){
         cout << s[i];
        }
        cout << endl;
    }

 return 0;
}
