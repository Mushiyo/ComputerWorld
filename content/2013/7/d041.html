解題雜記：
　　這題基本上就是用STL的sort，再加上自己自定義的排序function就好。寫排序function時要注意它的回傳值的意思是「參數1是否小於參數2」，這部份如果寫錯的話很可能會導致RE。我就是因為這個關係吃了一堆RE，而且很久都不明白為什麼會RE，直到看到這篇才終於結束RE惡夢……

C++： 
/*Filename: d041.cpp
 *Author: Mushiyo
 */

#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

bool lexicographicalCmp(const char &c1, const char &c2) {
 char lowerC1 = tolower(c1);
 char lowerC2 = tolower(c2);

 if (lowerC1 == lowerC2) {
  if (c1 >= 'a' && c2 < 'a') { // c1 between a - z, c2 between A - Z
   return true;
  } else {
   return false;
  }
 }

 return lowerC1 < lowerC2;
}

int main() {
 string s;

 while (getline(cin, s)) {
  int len = s.size();

  if (len == 0) {
   printf("\n");
   continue;
  }

  sort(s.begin(), s.end(), lexicographicalCmp);

  printf("%s\n", s.c_str());
 }

 return 0;
}
